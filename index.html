/* 請在 index.html 找到 handleGenerateSummary 並替換 */ const handleGenerateSummary = async () => { const apiKey = prompt("請輸入 Gemini API Key:"); if (!apiKey) return; setIsGenerating(true); setSummary(""); // 清空舊內容 try { const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${apiKey}`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: `你是一位專業的藥廠特助。請將以下 ${companyName} 的會議內容整理為一份正式公文：\n主題：${theme}\n日期：${date}\n內容詳情：${JSON.stringify(reports)}` }] }] }) }); const data = await response.json(); // 診斷邏輯：如果 API 有報錯，直接顯示原因 if (data.error) { throw new Error(`API 錯誤：${data.error.message} (代碼: ${data.error.status})`); } if (data.candidates && data.candidates[0] && data.candidates[0].content) { setSummary(data.candidates[0].content.parts[0].text); } else { throw new Error("AI 回傳了空內容，請檢查輸入的資訊是否過少。"); } } catch (e) { alert("系統診斷回報：\n" + e.message); console.error(e); } finally { setIsGenerating(false); } };
